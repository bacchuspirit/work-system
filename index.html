設定步驟

  1. 開啟你的 Google Sheet → 點上方「延伸功能」→「Apps Script」
  2. 貼上下方腳本，點「儲存」
  3. 點「部署」→「新增部署項目」→ 類型選「網頁應用程式」
  4. 執行身分：我；存取權限：所有人 → 部署 → 複製網址
  5. 將網址貼到上方輸入欄，點「儲存」
function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const { name, date, in: inTime, out } = e.parameter; if (!sheet.getLastRow() || sheet.getRange(1,1).getValue() === '') { sheet.appendRow(['姓名', '日期', '上班時間', '下班時間']); } const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == name && data[i][1] == date) { if (out) sheet.getRange(i+1, 4).setValue(out); return ContentService.createTextOutput('updated'); } } sheet.appendRow([name, date, inTime, out || '']); return ContentService.createTextOutput('ok'); }
